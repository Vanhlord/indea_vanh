CREATE TABLE mailbox_items (
    id TEXT PRIMARY KEY,                    -- UUID
    playerName TEXT NOT NULL,               -- Minecraft name
    playerDiscordId TEXT,                   -- Optional: Discord ID
    itemId TEXT NOT NULL,                   -- diamond_block, golden_apple, etc.
    quantity INTEGER NOT NULL DEFAULT 1,
    status TEXT DEFAULT 'pending',          -- pending, claimed, expired
    createdAt DATETIME DEFAULT CURRENT_TIMESTAMP,
    claimedAt DATETIME,
    expiresAt DATETIME,                     -- Auto-delete sau 30 ngày
    addedBy TEXT,                           -- Admin Discord ID
    notes TEXT
);

CREATE TABLE minecraft_items_config (
    itemId TEXT PRIMARY KEY,
    itemName TEXT,
    displayName TEXT,
    description TEXT,
    rarity TEXT                            -- common, rare, epic, legendary
);

CREATE TABLE mailbox_history (
    id TEXT PRIMARY KEY,
    itemId TEXT,
    playerName TEXT,
    action TEXT,                           -- 'added', 'claimed', 'expired'
    timestamp DATETIME DEFAULT CURRENT_TIMESTAMP,
    details TEXT
);

import axios from 'axios';

const PIKAMC_PANEL_URL = process.env.PIKAMC_PANEL_URL;
const PIKAMC_API_KEY = process.env.PIKAMC_API_KEY;
const PIKAMC_SERVER_ID = process.env.PIKAMC_SERVER_ID;

/**
 * Gửi lệnh đến Minecraft server qua Pterodactyl API
 */
export async function executeMinecraftCommand(command) {
    try {
        const url = `${PIKAMC_PANEL_URL}/api/client/servers/${PIKAMC_SERVER_ID}/command`;
        
        console.log(`🔨 [PIKAMC] Executing: ${command}`);
        
        const response = await axios.post(
            url,
            { command },
            {
                headers: {
                    'Authorization': `Bearer ${PIKAMC_API_KEY}`,
                    'Content-Type': 'application/json',
                    'Accept': 'application/json'
                }
            }
        );
        
        console.log('✅ [PIKAMC] Command executed successfully');
        return true;
    } catch (error) {
        console.error('❌ [PIKAMC] Command failed:', error.response?.data || error.message);
        return false;
    }
}

/**
 * Cho player một item
 */
export async function giveItemToPlayer(playerName, itemId, quantity = 1) {
    const command = `/give @p[name=${playerName}] ${itemId} ${quantity}`;
    return executeMinecraftCommand(command);
}

/**
 * Lấy danh sách player online (gửi lệnh /list)
 */
export async function getOnlinePlayers() {
    try {
        const url = `${PIKAMC_PANEL_URL}/api/client/servers/${PIKAMC_SERVER_ID}/command`;
        
        await axios.post(
            url,
            { command: '/list' },
            {
                headers: {
                    'Authorization': `Bearer ${PIKAMC_API_KEY}`,
                    'Content-Type': 'application/json'
                }
            }
        );
        
        return true;
    } catch (error) {
        console.error('Failed to get player list:', error.message);
        return false;
    }
}

import express from 'express';
import { v4 as uuid } from 'uuid';
import db from '../modules/database.js';
import { giveItemToPlayer, executeMinecraftCommand } from '../services/pikamcService.js';

const router = express.Router();

// ===== MIDDLEWARE: Check Admin =====
function isAdmin(userId) {
    const admins = process.env.ADMIN_ID?.split(',') || [];
    return admins.includes(String(userId));
}

// ===== ADD ITEM TO MAILBOX (Admin) =====
router.post('/add', async (req, res) => {
    const { playerName, itemId, quantity } = req.body;
    
    if (!isAdmin(req.user?.id)) {
        return res.status(403).json({ error: 'Admin only' });
    }
    
    // Validate item
    const validItems = ['diamond_block', 'golden_apple', 'diamond_sword', 'netherite_pickaxe', 'diamond_pickaxe', 'shield'];
    if (!validItems.includes(itemId)) {
        return res.status(400).json({ error: 'Invalid item ID' });
    }
    
    try {
        const itemUuid = uuid();
        db.run(`
            INSERT INTO mailbox_items (id, playerName, itemId, quantity, addedBy, status)
            VALUES (?, ?, ?, ?, ?, 'pending')
        `, [itemUuid, playerName, itemId, quantity || 1, req.user.id]);
        
        // Log to history
        db.run(`
            INSERT INTO mailbox_history (id, itemId, playerName, action, details)
            VALUES (?, ?, ?, 'added', ?)
        `, [uuid(), itemUuid, playerName, `Added by ${req.user.username}`]);
        
        res.json({ 
            success: true, 
            message: `✅ Added ${quantity || 1}x ${itemId} to ${playerName}'s mailbox` 
        });
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
});

// ===== GET MAILBOX (Player) =====
router.get('/:playerName', async (req, res) => {
    const playerName = req.params.playerName;
    
    try {
        const items = db.all(`
            SELECT id, itemId, quantity, createdAt, status
            FROM mailbox_items
            WHERE playerName = ? AND status = 'pending'
            ORDER BY createdAt DESC
        `, [playerName]);
        
        res.json({ 
            success: true,
            playerName,
            totalItems: items.length,
            items 
        });
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
});

// ===== CLAIM ITEM (Player) =====
router.post('/claim/:itemId', async (req, res) => {
    const itemId = req.params.itemId;
    const playerName = req.user.minecraftName;
    
    try {
        // Get item from DB
        const item = db.get(`
            SELECT * FROM mailbox_items 
            WHERE id = ? AND playerName = ?
        `, [itemId, playerName]);
        
        if (!item) {
            return res.status(404).json({ error: 'Item not found' });
        }
        
        if (item.status !== 'pending') {
            return res.status(400).json({ error: 'This item has already been claimed' });
        }
        
        // Execute /give command
        const success = await giveItemToPlayer(
            playerName, 
            item.itemId, 
            item.quantity
        );
        
        if (!success) {
            return res.status(500).json({ 
                error: 'Failed to give item in game. Make sure you are online!',
                action: 'retry_when_online'
            });
        }
        
        // Mark as claimed in DB
        db.run(`
            UPDATE mailbox_items 
            SET status = 'claimed', claimedAt = datetime('now')
            WHERE id = ?
        `, [itemId]);
        
        // Log to history
        db.run(`
            INSERT INTO mailbox_history (id, itemId, playerName, action)
            VALUES (?, ?, ?, 'claimed')
        `, [uuid(), itemId, playerName]);
        
        res.json({ 
            success: true, 
            message: `✅ Received ${item.quantity}x ${item.itemId}!` 
        });
        
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
});

// ===== ADMIN: EXECUTE COMMAND DIRECTLY =====
router.post('/admin/execute-command', async (req, res) => {
    if (!isAdmin(req.user?.id)) {
        return res.status(403).json({ error: 'Admin only' });
    }
    
    const { command } = req.body;
    
    try {
        const success = await executeMinecraftCommand(command);
        res.json({ 
            success, 
            message: success ? 'Command executed' : 'Command failed' 
        });
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
});

// ===== ADMIN: VIEW ALL MAILBOXES =====
router.get('/admin/all', async (req, res) => {
    if (!isAdmin(req.user?.id)) {
        return res.status(403).json({ error: 'Admin only' });
    }
    
    try {
        const items = db.all(`
            SELECT playerName, COUNT(*) as totalItems, 
                   SUM(CASE WHEN status='pending' THEN 1 ELSE 0 END) as pendingItems
            FROM mailbox_items
            GROUP BY playerName
            ORDER BY pendingItems DESC
        `);
        
        res.json({ success: true, data: items });
    } catch (error) {
        res.status(500).json({ error: error.message });
    }
});

export default router;

<script>
    async function loadMailbox() {
        try {
            const res = await fetch(`/api/mailbox/${playerName}`);
            const data = await res.json();
            
            document.getElementById('loadingState').classList.add('hidden');
            
            if (data.items && data.items.length > 0) {
                document.getElementById('mailboxGrid').classList.remove('hidden');
                renderItems(data.items);
            } else {
                document.getElementById('emptyState').classList.remove('hidden');
            }
        } catch (error) {
            console.error('Failed to load mailbox:', error);
        }
    }

    function renderItems(items) {
        const grid = document.getElementById('mailboxGrid');
        grid.innerHTML = items.map(item => `
            <div class="bg-gradient-to-br from-gray-800 to-gray-900 rounded-lg p-6 border border-gray-700 hover:border-blue-500 transition">
                <div class="flex justify-between items-start mb-4">
                    <div>
                        <h3 class="text-xl font-bold text-blue-400">${item.itemId}</h3>
                        <p class="text-gray-400">x${item.quantity}</p>
                    </div>
                    <i class="fas fa-box text-3xl text-yellow-400"></i>
                </div>
                <p class="text-xs text-gray-500 mb-4">${new Date(item.createdAt).toLocaleDateString('vi-VN')}</p>
                <button 
                    onclick="claimItem('${item.id}')"
                    class="w-full bg-gradient-to-r from-green-600 to-emerald-600 hover:from-green-700 hover:to-emerald-700 text-white font-bold py-2 rounded-lg transition"
                >
                    ✅ Nhận Hàng
                </button>
            </div>
        `).join('');
    }

    async function claimItem(itemId) {
        try {
            const res = await fetch(`/api/mailbox/claim/${itemId}`, { method: 'POST' });
            const data = await res.json();
            
            if (res.ok) {
                alert('✅ ' + data.message);
                loadMailbox(); // Reload
            } else {
                alert('❌ ' + data.error);
            }
        } catch (error) {
            alert('❌ Lỗi: ' + error.message);
        }
    }

    async function adminAddItem() {
        const playerName = document.getElementById('adminPlayerName').value;
        const itemId = document.getElementById('adminItemId').value;
        const quantity = parseInt(document.getElementById('adminQuantity').value) || 1;
        
        if (!playerName || !itemId) {
            alert('❌ Fill all fields');
            return;
        }
        
        try {
            const res = await fetch('/api/mailbox/add', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ playerName, itemId, quantity })
            });
            
            const data = await res.json();
            alert(data.message);
        } catch (error) {
            alert('❌ Error: ' + error.message);
        }
    }

    // Load on page load
    loadMailbox();
</script>
