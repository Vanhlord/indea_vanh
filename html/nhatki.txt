-- Draggable Handle
local function makeDraggable(frame, h)
    local d, di, ds, sp
    h.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            d = true; ds = input.Position; sp = frame.Position
            input.Changed:Connect(function() if input.UserInputState == Enum.UserInputState.End then d = false end end)
        end
    end)
    UserInputService.InputChanged:Connect(function(i)
        if d and (i.UserInputType == Enum.UserInputType.MouseMovement or i.UserInputType == Enum.UserInputType.Touch) then
            local delta = i.Position - ds
            frame.Position = UDim2.new(sp.X.Scale, sp.X.Offset + delta.X, sp.Y.Scale, sp.Y.Offset + delta.Y)
        end
    end)
end

local function updateBrightness()
    BrightnessLabel.Text = "NHÌN TRONG TỐI: " .. currentBrightnessLevel .. " / 20"
    Lighting.ExposureCompensation = originalExposure + ((currentBrightnessLevel / 20) * 2.5)
    local step = math.floor((currentBrightnessLevel / 20) * 150)
    Lighting.Ambient = Color3.fromRGB(step, step, step)
end

-- ==================== FLIGHT LOGIC ====================
RunService.Heartbeat:Connect(function()
    local char = LocalPlayer.Character
    if char and char:FindFirstChild("HumanoidRootPart") and char:FindFirstChild("Humanoid") then
        local hrp = char.HumanoidRootPart
        local hum = char.Humanoid
        
        if isFlyEnabled then
            if not flyBV then
                flyBV = Instance.new("BodyVelocity")
                flyBV.MaxForce = Vector3.new(1, 1, 1) * 10^6
                flyBV.Parent = hrp
            end
            if not flyBG then
                flyBG = Instance.new("BodyGyro")
                flyBG.MaxTorque = Vector3.new(1, 1, 1) * 10^6
                flyBG.P = 10^6
                flyBG.Parent = hrp
            end
            
            flyBG.CFrame = Camera.CFrame
            local dir = hum.MoveDirection
            flyBV.Velocity = dir * currentSpeedValue
            
            -- Keep altitude if not moving
            if dir.Magnitude == 0 then
                flyBV.Velocity = Vector3.new(0, 0, 0)
            end
        else
            if flyBV then flyBV:Destroy(); flyBV = nil end
            if flyBG then flyBG:Destroy(); flyBG = nil end
        end
    end
end)
